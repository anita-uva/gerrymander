cat ky.orig | grep -v Page | grep -v Totals | grep -v "Voter Registration Statistics Report" | grep -v "State Board of Elections" | grep -v ^Precinct | grep -v '^[0-9][0-9][0-9]' | awk  '{ if (NF < 13) printf("%s ",$0); else printf("%s\n",$0);  }' | awk '{ print $0,$1 }' | sed 's/^.....//' | sed -E -n 's/(^.*) ([0-9]\-)/"\1", \2/p'  | awk -F, '{ gsub(/[[:blank:]]/,",",$2); print }' > ky.csv
